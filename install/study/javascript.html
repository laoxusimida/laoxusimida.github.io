    # 子构造函数继承父构造函数方法

        // 借用父构造函数继承属性
    // 1 父构造函数
    function Father(uname, age) {
        this.uname = uname;
        this.age = age
    }
    // 子构造函数  必须将父构造函数的this也继承过来才可以使用父构造函数内的属性以及方法
    function Son(uname, age, score) {
        Father.call(this,uname, age)
        this.score = score
    }
    var son = new Son('小徐', 18, 98)
    console.log(son);

    # http与https的区别

    http与https区别
http为明文传输会被抓包窃取
https=http+ssl/tsl 安全套接层用来传输数据加密
对称数加密：客户端和服务器通过同一把钥匙进行加密解密---钥匙被黑客获取不安全
非对称加密：客户端进行公钥加密，服务端通过私钥进行解密
客户端接收服务端返回的公钥可能被中间人拦截，返回一个虚假的公钥会将客户端传递的数据暴露
此时需要证书签名和公钥共同使用，在返回公钥时对证书签名进行验证才知道公钥是不是正确的

真正的https传输是通过非对称式加密进行建立安全关系，服务端有公钥和私钥以及秘钥，之后的通信
通过秘钥进行对称式加密进行的
